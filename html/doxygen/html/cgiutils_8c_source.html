02477 /*
02478  * copy a file from the path at source to the already opened
02479  * destination file dest.
02480  * This is handy when creating tempfiles with mkstemp()
02481  */
02482 int my_fdcopy(char *source, char *dest, int dest_fd){
02483         int source_fd, rd_result = 0, wr_result = 0;
02484         unsigned long tot_written = 0, tot_read = 0, buf_size = 0;
02485         struct stat st;
02486         char *buf;
02487 
02488         /* open source file for reading */
02489         if((source_fd=open(source,O_RDONLY,0644)) < 0){
02490                 logit(NSLOG_RUNTIME_ERROR,TRUE,"Error: Unable to open file '%s' for reading: %s\n",source,strerror(errno));
02491                 return ERROR;
02492         }
02493 
02494         /*
02495          * find out how large the source-file is so we can be sure
02496          * we've written all of it
02497          */
02498         if (fstat(source_fd, &st) < 0) {
02499                 logit(NSLOG_RUNTIME_ERROR,TRUE,"Error: Unable to stat source file '%s' for my_fcopy(): %s\n", source, strerror(errno));
02500                 close(source_fd);
02501                 return ERROR;
02502         }
02503 
02504         /*
02505          * If the file is huge, read it and write it in chunks.
02506          * This value (128K) is the result of "pick-one-at-random"
02507          * with some minimal testing and may not be optimal for all
02508          * hardware setups, but it should work ok for most. It's
02509          * faster than 1K buffers and 1M buffers, so change at your
02510          * own peril. Note that it's useful to make it fit in the L2
02511          * cache, so larger isn't necessarily better.
02512          */
02513         buf_size = st.st_size > 128 << 10 ? 128 << 10 : st.st_size;
02514         buf = malloc(buf_size);
02515         if (!buf) {
02516                 logit(NSLOG_RUNTIME_ERROR,TRUE,"Error: Unable to malloc(%lu) bytes: %s\n", buf_size, strerror(errno));
02517                 close(source_fd);
02518                 return ERROR;
02519         }
02520         /* most of the times, this loop will be gone through once */
02521         while (tot_written < st.st_size) {
02522                 int loop_wr = 0;
02523 
02524                 rd_result = read(source_fd, buf, buf_size);
02525                 if (rd_result < 0) {
02526                         if (errno == EAGAIN || errno == EINTR)
02527                                 continue;
02528                         logit(NSLOG_RUNTIME_ERROR,TRUE,"Error: my_fcopy() failed to read from '%s': %s\n", source, strerror(errno));
02529                         break;
02530                 }
02531                 tot_read += rd_result;
02532 
02533                 while (loop_wr < rd_result) {
02534                         wr_result = write(dest_fd, buf + loop_wr, rd_result - loop_wr);
02535 
02536                         if (wr_result < 0) {
02537                                 if (errno == EAGAIN || errno == EINTR)
02538                                         continue;
02539                                 logit(NSLOG_RUNTIME_ERROR,TRUE,"Error: my_fcopy() failed to write to '%s': %s\n", dest, strerror(errno));
02540                                 break;
02541                         }
02542                         loop_wr += wr_result;
02543                 }
02544                 if (wr_result < 0)
02545                         break;
02546                 tot_written += loop_wr;
02547         }
02548 
02549         /*
02550          * clean up irregardless of how things went. dest_fd comes from
02551          * our caller, so we mustn't close it.
02552          */
02553         close(source_fd);
02554         free(buf);
02555 
02556         if (rd_result < 0 || wr_result < 0) {
02557                 /* don't leave half-written files around */
02558                 unlink(dest);
02559                 return ERROR;
02560         }
02561 
02562         return OK;
02563 }
